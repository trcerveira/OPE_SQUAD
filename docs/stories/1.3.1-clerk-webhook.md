## Technical Notes

### Reuse from OPB Crew
The OPB Crew already has `syncUserProfile()` in `lib/supabase/user-profiles.ts`. This webhook calls that function on Clerk events.

### Clerk Webhook Setup
1. In Clerk Dashboard → Webhooks → Add Endpoint
2. URL: `https://publyq.ai/api/webhooks/clerk`
3. Events: `user.created`, `user.updated`
4. Copy signing secret → `CLERK_WEBHOOK_SECRET` env var

### Implementation Pattern
```typescript
import { Webhook } from 'svix';
import { WebhookEvent } from '@clerk/nextjs/server';

export async function POST(req: Request) {
  const payload = await req.text();
  const headers = Object.fromEntries(req.headers);

  const wh = new Webhook(process.env.CLERK_WEBHOOK_SECRET!);
  const evt = wh.verify(payload, headers) as WebhookEvent;

  if (evt.type === 'user.created' || evt.type === 'user.updated') {
    await syncUserProfile({
      clerkId: evt.data.id,
      email: evt.data.email_addresses[0]?.email_address,
      name: `${evt.data.first_name || ''} ${evt.data.last_name || ''}`.trim(),
    });
  }

  return new Response('OK', { status: 200 });
}
```
